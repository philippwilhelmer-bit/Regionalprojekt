The dispatcher mirrors the existing `src/lib/ingestion/adapters/registry.ts` pattern (keyed off the `ArticleSource` Prisma enum) so source-type routing stays consistent across the codebase.

Purpose: Closes AIPL-06. Today the pipeline uses `JSON.stringify(rawPayload)` (`pipeline.ts:115-117`) which sends `EMITTENT`, `WEBLINK`, `Rückfragen`, contact phone numbers, and email addresses into the LLM input. This is the root cause of source-metadata bleed into article text. Clean extractors strip the metadata at the source-of-truth boundary so the prompt's "no Presseformular-Floskeln" rule (which is the second line of defense) is never the only line.

Output: A self-contained extractor package with full unit coverage. NO pipeline modification in this plan — wiring happens in 43-03. This plan is wave-1 parallel with 43-01 because it shares no files with the merged-call work.

```typescript
// Required behavior (sketch — adapt to real CANDIDATE_BODY_FIELDS contents you discover):
const OTS_METADATA_LINE_PATTERNS = [
  /Rückfragen\s*&?\s*Kontakt/i,
  /Pressekontakt/i,
  /^Aussender/im,
  /Tel\.?:\s*[+0-9 ()\\/-]+/i,
  /E-?Mail:?\s*\S+@\S+/i,
  /\bOTS\d{3,4}\b/i,
  // websites — naïve catch (e.g. "https://...", "www....")
]

export function extractOts(rawPayload: unknown, title?: string, content?: string): string {
  if (!rawPayload || typeof rawPayload !== 'object') {
    return [title, content].filter(Boolean).join('\n\n')
  }
  const detail = rawPayload as Record<string, unknown>
  const otsTitle = typeof detail['TITEL'] === 'string' ? detail['TITEL'] : (title ?? '')
  const otsBody = extractBodyFromOts(detail) || (content ?? '')
  const cleanedBody = stripMetadataLines(otsBody)
  return [otsTitle, cleanedBody].filter(Boolean).join('\n\n').trim()
}

function extractBodyFromOts(detail: Record<string, unknown>): string {
  // Use the EXACT CANDIDATE_BODY_FIELDS list from src/lib/ingestion/adapters/ots-at.ts
  // (read it at task start; do not invent the list).
  for (const field of CANDIDATE_BODY_FIELDS) {
    const v = detail[field]
    if (typeof v === 'string' && v.trim().length > 0) return v
  }
  return ''
}

function stripMetadataLines(body: string): string {
  return body
    .split(/\r?\n/)
    .filter((line) => !OTS_METADATA_LINE_PATTERNS.some((re) => re.test(line)))
    .join('\n')
    .replace(/\n{3,}/g, '\n\n')
    .trim()
}
```

**`src/lib/ai/extractors/rss.ts`:**
```typescript
export function extractRss(rawPayload: unknown, title?: string, content?: string): string {
  if (!rawPayload || typeof rawPayload !== 'object') {
    return [title, content].filter(Boolean).join('\n\n')
  }
  const item = rawPayload as Record<string, unknown>
  const rssTitle = typeof item['title'] === 'string' ? item['title'] : (title ?? '')
  const body =
    (typeof item['description'] === 'string' ? item['description'] : null) ??
    (typeof item['summary'] === 'string' ? item['summary'] : null) ??
    (typeof item['content'] === 'string' ? item['content'] : (content ?? ''))
  return [rssTitle, body].filter(Boolean).join('\n\n').trim()
}
```

**`src/lib/ai/extractors/index.ts`:**
```typescript
import type { ArticleSource } from '@prisma/client'
import { extractOts } from './ots'
import { extractRss } from './rss'

export type ExtractorFn = (rawPayload: unknown, title?: string, content?: string) => string

export const extractorRegistry: Partial<Record<ArticleSource, ExtractorFn>> = {
  OTS_AT: extractOts,
  RSS: extractRss,
  // MANUAL: intentionally absent — falls through to default
}

export function extractArticleText(
  source: ArticleSource,
  rawPayload: unknown,
  title?: string,
  content?: string,
): string {
  const extractor = extractorRegistry[source]
  if (extractor) return extractor(rawPayload, title, content)
  return [title, content].filter(Boolean).join('\n\n')
}
```
