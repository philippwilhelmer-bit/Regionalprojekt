       ```typescript
       /**
        * Run Nominatim geocoding + static map generation for a doctor address.
        * Both calls are independent try/catch'd — neither blocks the save.
        * @returns { geo, mapImageUrl } — either may be null on failure
        */
       async function geocodeAndMap(
         address: string,
         name: string,
         doctorId: number,
       ): Promise<{ geo: GeoResult | null; mapImageUrl: string | null }> {
         let geo: GeoResult | null = null
         try {
           const result = await geocodeLocation(defaultPrisma, address)
           if (result) {
             geo = { lat: result.lat, lon: result.lon, locationType: result.locationType }
           }
         } catch (err) {
           console.warn(`[doctors-actions] geocode failed for "${address}": ${String(err)}`)
         }

         let mapImageUrl: string | null = null
         if (geo) {
           try {
             const map = await generateMapImage(
               geo.lat,
               geo.lon,
               name,
               doctorId,
               geo.locationType,
               { pathPrefix: 'doctor' },
             )
             if (map) mapImageUrl = map.url
           } catch (err) {
             // generateMapImage already catches internally and returns null,
             // but defensive try/catch in case sharp/blob library surface a new error class.
             console.warn(`[doctors-actions] mapgen threw for doctor=${doctorId}: ${String(err)}`)
           }
         }

         return { geo, mapImageUrl }
       }
       ```

       Note the `doctorId` paradox: at create-time we don't have an ID yet. Resolve via the **two-phase create**: insert row first with null geo, get the ID, then geocode/map, then update with geo+map. This is the cleanest pattern given the Blob path needs the ID.

    2. **createDoctor Server Action** (two-phase: insert → geocode+map → update):

       ```typescript
       export async function createDoctor(input: CreateDoctorInput): Promise<Doctor> {
         await requireAuth()
         // Phase 1: create row with null geo so we have a stable ID for the map Blob path
         const created = await createDoctorDb(defaultPrisma, input)

         const { geo, mapImageUrl } = await geocodeAndMap(input.address, input.name, created.id)

         return created
       }
       ```

    3. **updateDoctor Server Action** (re-geocode iff address changed):

       ```typescript
       export async function updateDoctor(input: UpdateDoctorInput): Promise<Doctor> {
         await requireAuth()
         // Read previous row to compare address
         const prev = await defaultPrisma.doctor.findUniqueOrThrow({ where: { id: input.id } })
         const addressChanged = input.address !== undefined && input.address !== prev.address

         if (!addressChanged) {
           return updateDoctorDb(defaultPrisma, input)
         }
         // Address changed → re-geocode (single call, no sleep needed)
         const { geo, mapImageUrl } = await geocodeAndMap(input.address!, input.name ?? prev.name, input.id)
         return updateDoctorDb(defaultPrisma, input, geo ?? null, mapImageUrl ?? null)
       }
       ```
