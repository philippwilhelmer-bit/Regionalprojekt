       ```typescript
       import { notFound } from 'next/navigation'
       import { listBezirke } from '@/lib/content/bezirke'
       import { getDoctorById } from '@/lib/content/doctors'
       import { updateDoctorForm } from '@/lib/admin/doctors-actions'
       import DoctorForm from '../../DoctorForm'

       export const dynamic = 'force-dynamic'

       type Props = { params: Promise<{ id: string }> }

       export default async function EditDoctorPage({ params }: Props) {
         const { id } = await params
         const doctorId = Number(id)
         if (!Number.isFinite(doctorId)) notFound()
         const [doctor, bezirke] = await Promise.all([
           getDoctorById(doctorId),
           listBezirke(),
         ])
         if (!doctor) notFound()
         return (
           <main className="...">
             <h1>Arzt bearbeiten: {doctor.name}</h1>
             <DoctorForm doctor={doctor} bezirke={bezirke} formAction={updateDoctorForm} submitLabel="Speichern" />
           </main>
         )
       }
       ```

    Both pages: `export const dynamic = 'force-dynamic'` (AGENTS.md rendering rule). NO `requireAuth()` — the `(admin)/layout.tsx` route group enforces auth for all children (confirmed at plan-revision time).
