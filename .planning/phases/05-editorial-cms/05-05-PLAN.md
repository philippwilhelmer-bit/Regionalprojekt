<!-- step2-write.ts prompt builder approach -->
Replace SYSTEM_PROMPT_TEMPLATE with a buildSystemPrompt() function:
```typescript
import { getResolvedAiConfig, type ResolvedAiConfig } from '../../admin/ai-config-dal'

function buildSystemPrompt(config: ResolvedAiConfig, bezirkContext: string): string {
  const toneInstruction = {
    NEUTRAL: 'Schreibe im sachlichen, neutralen Stil.',
    FORMAL: 'Schreibe in einem formellen, seriösen Stil.',
    CONVERSATIONAL: 'Schreibe in einem zugänglichen, gesprächigen Stil.',
  }[config.tone]

  const lengthInstruction = {
    SHORT: 'Der Artikel sollte 100–150 Wörter umfassen.',
    MEDIUM: 'Der Artikel sollte 150–250 Wörter umfassen.',
    LONG: 'Der Artikel sollte 250–400 Wörter umfassen.',
  }[config.articleLength]

  const styleNotes = config.styleNotes
    ? `\n\nZusätzliche Anweisungen: ${config.styleNotes}`
    : ''

  return `Du bist ein Redakteur für regionale österreichische Nachrichten...
Bezirk-Kontext: ${bezirkContext}
${toneInstruction}
${lengthInstruction}${styleNotes}`
}
```
