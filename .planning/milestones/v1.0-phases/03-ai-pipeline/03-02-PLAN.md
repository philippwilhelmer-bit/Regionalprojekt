Anthropic SDK structured output pattern (from RESEARCH.md):
```typescript
import Anthropic from '@anthropic-ai/sdk'

const response = await client.messages.create({
  model: 'claude-haiku-4-5-20251001',
  max_tokens: 256,
  system: systemPrompt,
  messages: [{ role: 'user', content: articleText }],
  output_config: {
    format: {
      type: 'json_schema',
      schema: {
        type: 'object',
        properties: {
          bezirkSlugs: { type: 'array', items: { type: 'string' } },
          hasNamedPerson: { type: 'boolean' }
        },
        required: ['bezirkSlugs', 'hasNamedPerson'],
        additionalProperties: false
      }
    }
  }
})

// Safe text block extraction
const textBlock = response.content.find(b => b.type === 'text')
if (!textBlock || textBlock.type !== 'text') throw new Error('No text block in response')
const result = JSON.parse(textBlock.text) as { bezirkSlugs: string[]; hasNamedPerson: boolean }

// Token counts always present
const inputTokens = response.usage.input_tokens
const outputTokens = response.usage.output_tokens
```

function buildBezirkContext(bezirke: Array<{ slug: string; name: string; gemeindeSynonyms: string[] }>): string {
  return bezirke.map(b =>
    `${b.slug} (${b.name}): ${b.gemeindeSynonyms.join(', ')}`
  ).join('\n')
}
