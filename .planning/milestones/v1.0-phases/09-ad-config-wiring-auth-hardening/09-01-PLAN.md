**Step 3 — Rewrite AdUnit.tsx as Server Component wrapper.**

`src/components/reader/AdUnit.tsx` — NO `"use client"` directive at top.

```typescript
import config from '@/../bundesland.config'
import { AdUnitClient } from './AdUnitClient'

type AdZone = 'hero' | 'between-articles' | 'article-detail'
interface AdUnitProps { zone: AdZone }

export function AdUnit({ zone }: AdUnitProps) {
  if (!config.features.ads) return null

  const zoneConfig = config.adZones.find(z => z.id === zone)
  if (!zoneConfig || !zoneConfig.enabled) return null

  const pubId = process.env.NEXT_PUBLIC_ADSENSE_PUB_ID
  const slot = process.env[zoneConfig.envVar]

  return <AdUnitClient pubId={pubId} slot={slot} zone={zone} />
}
```

**Critical constraints (from research — DO NOT violate):**
- AdUnit.tsx must NOT have `"use client"` — Server Component reads config at import time
- AdUnitClient must NOT be exported from AdUnit.tsx (call sites use named import `{ AdUnit }` only)
- `await requireAuth()` must NOT appear here — this is a public reader component, not an admin action
- `process.env[zoneConfig.envVar]` (NOT `process.env.NEXT_PUBLIC_*`) — server-only vars for slots; pubId stays NEXT_PUBLIC_ since AdUnitClient is a client component
